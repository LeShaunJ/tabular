    class Tabular::Habit

    #command(name : String, aliases = [] of String, help = "", directives : Directable | Nil = nil#

    Create a Command-flavoured Tablet along with its nested formation:

    Tabular.form do
  command "cmd1", help: "command with dispatched completions"

  command "cmd2", help: "command with inline completions" do
    option "--file", "-f" { argument }
    option "--debug"
    command "sub1"
    command "arg2_choice1", "arg2_choice2"
  end

  # will never trigger on `cmd2`
  dispatch do |command|
    Command1.complete if command.name == "cmd1"
  end
end
    • name: See Tablet#name.
    • aliases: See Tablet#aliases.
    • help: See Tablet#help.
    • directives: See Directive.

    #delimiters(value : String)#

    Specify the global string of characters that may delimit an Option-falvour Tablet.

    Tabular.form do
  # Allows for something like `--option=`
  delimiters "="
  # Allows for something like `-option:`
  delimiters ":"
  # Allows for all of the above
  delimiters ":="
end

    #installer(name = "completion", help = "install [TAB] completions")#

    Create a Tablet for the Command that installs completions on your users' shell.

    if Tabular.prompt?
  Tabular.form do
    command "cmd1"
    command "cmd2"

    # will complete `setup-tab` and possible params
    installer "setup-tab"

    dispatch do |command|
      # handle `cmd1` & `cmd2`
    end
  end
end

- *name*: See [`Tablet#name`][Tabular::Tablet#name].
- *help*: See [`Tablet#help`][Tabular::Tablet#help].
